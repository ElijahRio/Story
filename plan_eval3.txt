  // Store { text, dictionaryVersion, links }
  // How to track dictionaryVersion?
  // We can hash the names of all entities! Or just keep a single string of all names joined.

Let's do this:
Create a `dictionaryHash` which is `entities.map(e => e.name).join('|')`.
```javascript
  // ⚡ Bolt: Create a stable hash of all entity names to track when global link invalidation is required
  const dictionaryHash = useMemo(() => {
    return entities.map(e => e.name).join('|');
  }, [entities]);

  // ⚡ Bolt: Memoize text-based link detection per entity.
  // O(N^2) matching only runs fully when an entity's name changes (invalidating all)
  // or runs for a single entity when only its text fields change.
  const textLinksCacheRef = useRef({ hash: '', cache: new Map() });

  const textBasedLinks = useMemo(() => {
    const { hash, cache } = textLinksCacheRef.current;
    const isFullRecalculation = hash !== dictionaryHash;

    if (isFullRecalculation) {
      cache.clear();
      textLinksCacheRef.current.hash = dictionaryHash;
    }

    const currentLinks = [];

    entities.forEach(entity => {
      const allText = [
        entity.description,
        entity.systemic_inputs,
        entity.systemic_outputs,
        entity.biological_alterations,
        entity.attributes,
        entity.liabilities,
        entity.involved_records,
        entity.systemic_impact,
        entity.unresolved_threads,
        entity.ai_analysis
      ].filter(Boolean).join(' ');

      let cached = cache.get(entity.id);
      if (cached && cached.text === allText && !isFullRecalculation) {
         cached.links.forEach(targetId => {
           currentLinks.push({ source: entity.id, target: targetId, weight: 1.5, type: 'text' });
         });
         return;
      }

      // Compute
      const links = getDetectedLinks(allText, entity.id);
      const linkIds = links.map(l => l.id);

      cache.set(entity.id, { text: allText, links: linkIds });

      linkIds.forEach(targetId => {
        currentLinks.push({ source: entity.id, target: targetId, weight: 1.5, type: 'text' });
      });
    });

    return currentLinks;
  }, [entities, getDetectedLinks, dictionaryHash]);
```
Then in the `useEffect`:
```javascript
    // 1. Text-based references
    textBasedLinks.forEach(link => {
      addLinkWeight(link.source, link.target, link.weight, link.type);
    });
```
